#include <stdio.h>

int main()
{
    int n,a[100],i,min=9999;

    scanf("%d",&n);

    for(i=0;i<n;i++)
        scanf("%d",&a[i]);

    for(i=0;i<n;i++)
        if(a[i]>0 && a[i]<min)
            min=a[i];

    if(min==9999)
        printf("No positive");
    else
        printf("%d",min);

    return 0;
}
